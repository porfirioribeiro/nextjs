Function.prototype.Extends(Object baseClass) : 

!!Attention!!
This function is applied to class functions not methods
Extends this from other class
must be called right after constructor
function A() { }
A.prototype.name="A";
A.prototype.test=function() {
alert(this.name);
};
function B() { }
B.Extends(A);
B.prototype.name="B";

var a= new A(), b= new B();
a.test();//alerts "A"
b.test();//alerts "B"
